#Para asociar libros a autores....

>>>autor = Autor.objects.create(autor='R.A. Salvatore',nacionalidad='Norteamericano')

>>>libro = Libro.objects.create(titulo='El elfo oscuro: la morada',fecha_publicacion='1990-08-10',idioma='EN')
